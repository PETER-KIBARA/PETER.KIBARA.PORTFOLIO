    class Bird(pygame.sprite.Sprite):
    images = [pygame.image.load('bluebird-upflap.png'),
              pygame.image.load('bluebird-midflap.png'),
              pygame.image.load('bluebird-downflap.png')]

    def __init__(self):
        pygame.sprite.Sprite.__init__(self)
        self.rot = 0
        self.vel = 0
        self.birdIndex = 0
        self.image = self.images[self.birdIndex]
        self.rect = self.image.get_rect(center=(100, int(SCREENHEIGHT/2)))
        self.flapCycle = 0
        self.dropped = False
        self.rotCycle = 0
        self.rot = 0
        self.score = 0

    def update(self):
        self.vel += GRAVITY
        self.rect.centery += self.vel
        if self.rect.bottom > SCREENHEIGHT:
            self.rect.bottom = SCREENHEIGHT
            self.dropped = True
        if self.flapCycle == 5:
            self.birdIndex += 1
            if self.birdIndex == 3:
                self.birdIndex = 0
            self.flapCycle = 0
        self.image = pygame.transform.rotate(self.images[self.birdIndex], self.rot)
        self.flapCycle += 1

        if not self.dropped:
            if self.vel < 0:
                self.rot = 45
            elif self.vel > 0 and self.rot > -90:
                self.rot -= 5
        else:
            if self.rot > -90:
                self.rot -= 5

    def flap(self):
        if not self.dropped:
            self.vel = FLAPVELOCITY
            self.flapCycle = 0
            self.birdIndex = 0

    def score(self):
        self.score += 1


class Pipe(pygame.sprite.Sprite):
    def __init__(self, x, y, position):
        pygame.sprite.Sprite.__init__(self)
        self.image = pygame.image.load('pipe.png')
        self.rect = self.image.get_rect(topleft=(x, y))
        self.passed = False

    def update(self):
        self.rect.left -= SPEED

    def pass_bird(self, bird):
        if self.rect.right < bird.rect.left and not self.passed:
            self.passed = True
            return True
        return False


class Scoreboard(pygame.sprite.Sprite):
    def __init__(self, x, y):
        pygame.sprite.Sprite.__init__(self)
        self.score = 0
        self.font = pygame.font.SysFont('Comic Sans MS', 30)
        self.image = self.font.render(str(self.score), False, (0, 0, 0))
        self.rect = self.image.get_rect(topleft=(x, y))

    def update(self, score):
        self.score = score
        self.image = self.font.render(str(self.score), False, (0, 0, 0))
        self.rect = self.image.get_rect(topleft=self.rect.topleft)


def main():
    pygame.init()
    screen = pygame.display.set_mode((SCREENWIDTH, SCREENHEIGHT))
    pygame.display.set_caption('Flappy Bird')
    background = pygame.image.load('background.png')
    bird = Bird()
    pipes = pygame.sprite.Group()
    scoreboard = Scoreboard(10, 10)

    while True:
        for event in pygame.event.get():
            if event.type == QUIT:
                pygame.quit()
                sys.exit()
            if event.type == KEYDOWN:
                if event.key == K_SPACE:
                    bird.flap()

        pipes.update()
        bird.update()

        if len(pipes) == 0 or pipes.sprites()[-1].rect.right < SCREENWIDTH - PIPE_SPACING:
            pipe_pos = random.randint(PIPE_GAP_MIN, PIPE_GAP_MAX)
            pipes.add(Pipe(SCREENWIDTH, 0, 'top'))
            pipes.add(Pipe(SCREENWIDTH, SCREENHEIGHT - pipe_pos, 'bottom'))

        for pipe in pipes:
            if pipe.rect.right < 0:
                pipes.remove(pipe)
            if pipe.pass_bird(bird):
                bird.score()

        scoreboard.update(bird.score)

        screen.blit(background, (0, 0))
        pipes.draw(screen)
        screen.blit(bird.image, bird.rect)
        screen.blit(scoreboard.image, scoreboard.rect)

        if pygame.sprite.spritecollideany(bird, pipes) or bird.rect.bottom == SCREENHEIGHT:
            print("Game Over! Score:", bird.score)
            pygame.quit()
            sys.exit()

        pygame.display.update()


if __name__ == '__main__':
    main()
